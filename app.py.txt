st.write("Wähle deine wöchentlichen Aufgaben aus der Liste oder füge eigene hinzu.")

# Vordefinierte Aufgabenliste
default_tasks = [

    "Automatisierte Funnels & E-Mail-Marketing aufsetzen", "Vernetzen mit Profilen", "Termine setten im Chat",
    "Qualifizierungstelefonate führen", "Sales Calls führen", "Angebote versenden", "Sales Calls auswerten",

# Session State initialisieren
if "tasks" not in st.session_state:
    st.session_state.tasks = default_tasks.copy()
if "selected_tasks" not in st.session_state:
    st.session_state.selected_tasks = []

# Auswahl der Aufgaben
st.write("## Wähle deine Aufgaben aus")
for task in st.session_state.tasks:
    if st.checkbox(task, value=(task in st.session_state.selected_tasks)):
        if task not in st.session_state.selected_tasks:
            st.session_state.selected_tasks.append(task)
    else:
        if task in st.session_state.selected_tasks:
            st.session_state.selected_tasks.remove(task)

# Eigene Aufgaben hinzufügen
st.write("## Eigene Aufgaben hinzufügen")
new_task = st.text_input("Neue Aufgabe eingeben und auf 'Hinzufügen' klicken:")
if st.button("Hinzufügen"):
    if new_task and new_task not in st.session_state.tasks:
        st.session_state.tasks.append(new_task)
        st.experimental_rerun()

# Anzeige der gewählten Aufgaben
st.write("## Deine ausgewählten Aufgaben")
if st.session_state.selected_tasks:
    for task in st.session_state.selected_tasks:
        st.write(f"✅ {task}")
else:
    st.write("Noch keine Aufgaben ausgewählt.")
